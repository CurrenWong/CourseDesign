// 登陆接口
// 功能：输入用户名和密码，查询是否与数据库中的信息匹配
// 若登陆成功，则返回用户的id，并将isLogin设为true
// 若用户名错误，则将isUsernameError设为true
// 若密码错误，则将isPasswordError设为true

// 输入:JSON格式
{
    // 用户名 (int),
    "username": 123456,
    // 密码 (int)
    "password": 123456,
    // 身份 (string) 为四个其中的一个["student", "university", "education_department", "manager"]
    "identity": "education_department"
}

// 输出:JSON格式
{
    // 是否登陆成功，默认为false
    "isLogin": true,
    // 用户ID，默认为NULL
    "userId": 1,
    // 标示是否用户名错误，默认为false
    "isUsernameError": false,
    // 标示是否密码错误，默认为false
    "isPasswordError": false
}
